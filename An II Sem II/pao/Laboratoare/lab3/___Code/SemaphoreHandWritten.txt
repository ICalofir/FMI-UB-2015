class Semaphore
{
  private int mVal = 0;
  public Semaphore(int initial) { mVal = initial; }
  public synchronized void myWait() 
  {
    mVal--;
    if (mVal < 0) try { wait(); } 
    catch(InterruptedException e) { }
  }

  public synchronized void mySignal() 
  {
    mVal++;
    if (mVal <=0 ) notify();
  }
}
